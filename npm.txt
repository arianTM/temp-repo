//  npm: node package manager
//  package = module = dependency

//  npm - global command, comes with node
//    npm --version

//  local dependency - use it only in this folder
//    npm i <packageName>

//  global dependency - use it in any folder
//    npm install -g <packageName>
//    sudo npm install -g <packageName> (macOS)


//  package.json - manifest file (stores INFO about the packages)

//  -> manual approach: create 'package.json' in the root, add properties to it
//  -> npm init (step by step, press ENTER to SKIP)
//  -> npm init -y (all default)

// node_modules - folder that stores all the packages code
